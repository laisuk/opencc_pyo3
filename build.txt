create module folder in project root /opencc_pyo3 same name as sated in cargo.toml:
[lib]
name = "opencc_pyo3"

create pyproject.toml

prepare files in module folder:
__init__.py
opencc_pyo3.pyi

prepare .venv if needed

maturin develop -r (Create pyd to opencc_pyo3 folder, need venv)
maturin build -r

Windows:
pip install ./target/wheels/opencc_pyo3-1.0.0-cp312-none-win_amd64.whl --force-reinstall

pip install ./target/wheels/opencc_pyo3-1.0.0-cp312-cp312-manylinux_2_34_x86_64.whl --break-system-packages --force-reinstall
